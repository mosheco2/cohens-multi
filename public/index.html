// server.js const express = require('express'); const path = require('path'); const app = express(); // Render נותן PORT כ־env variable const PORT = process.env.PORT || 10000; // סטטיקה מהתיקייה public app.use(express.static(path.join(__dirname, 'public'))); // דף הבית app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); // דף מנהל app.get('/host', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'host.html')); }); // דף שחקן app.get('/player', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'player.html')); }); // דף הוראות app.get('/instructions', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'instructions.html')); }); // דף ניהול באנרים (אם תרצה) app.get('/admin-banners', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'admin-banners.html')); }); app.listen(PORT, () => { console.log(`Server listening on port ${PORT}`); });
